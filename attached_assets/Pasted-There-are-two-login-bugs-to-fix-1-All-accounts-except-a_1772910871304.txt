Find where the vendor name is displayed. It should read from the JWT payload like this:

STEP 8 — Rebuild fresh password hashes for all test accounts

Run this to generate and apply fresh hashes:

python3 << 'EOF'
import asyncio
from passlib.context import CryptContext
from sqlalchemy import text
from app.database import engine

pwd = CryptContext(schemes=['bcrypt'], deprecated='auto')

accounts = [
    ('admin@bowenstreetmarket.com', 'admin123'),
    ('cashier@bowenstreetmarket.com', 'cashier123'),
    ('sarah@email.com', 'vendor123'),
    ('mike@email.com', 'vendor123'),
    ('linda@email.com', 'vendor123'),
    ('nora@email.com', 'vendor123'),
    ('sammy@email.com', 'vendor123'),
    ('ashley@email.com', 'vendor123'),
    ('anne@email.com', 'vendor123'),
    ('paula@email.com', 'vendor123'),
]

async def reset():
    async with engine.begin() as conn:
        for email, password in accounts:
            h = pwd.hash(password)
            await conn.execute(
                text("UPDATE vendors SET password_hash = :h, status = 'active' WHERE email = :e"),
                {"h": h, "e": email}
            )
            print(f"Reset: {email}")

asyncio.run(reset())
EOF
